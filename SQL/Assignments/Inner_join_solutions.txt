2.

FROM EMP INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO

3. 

SELECT EMPNO, SAL, LOC
FROM EMP INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE JOB = 'SALESMAN'

4. 

SELECT EMPNO, SAL, COMM, LOC 
FROM EMP INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.SAL > 2000 


5. 

SELECT ENAME, JOB, LOC 
FROM EMP INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.JOB = 'PRESIDENT' 
AND DEPT.DNAME = 'ACCOUNTING'

6. 

SELECT EMPNO, SAL, DNAME 
FROM EMP INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.COMM IS NOT NULL
AND EMP.JOB = 'SALESMAN'

7.

FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO

8.  

SELECT DEPT.* , EMP.ENAME
FROM EMP
INNER JOIN DEPT

9. 

SELECT EMP.*, DEPT.*
FROM EMP
INNER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.JOB IN ('SALESMAN','MANAGER') AND DEPT.DNAME IN ('SALES','RESEARCH')


10 . 

SELECT EMP.ENAME, EMP.SAL, EMP.DEPTNO, DEPT.LOC
FROM EMP
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.JOB = 'ANALYST' AND DEPT.LOC IN ('DALLAS', 'BOSTON')


11.  

SELECT EMP.ENAME, EMP.COMM, EMP.HIREDATE, DEPT.DNAME, DEPT.LOC
FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.DEPTNO IN (20,30) AND EMP.JOB IN ('CLERK', 'MANAGER', 'SALESMAN')


12. 

SELECT SUM(EMP.SAL), DEPT.DNAME
FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DEPT.DNAME

13. 

SELECT MAX(EMP.SAL), DEPT.DNAME , COUNT(EMP.ENAME)
FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.JOB != 'PRESIDENT'
GROUP BY DEPT.DNAME
HAVING COUNT(EMP.ENAME) = 3

14. 


SELECT SUM(EMP.SAL), DEPT.DNAME, COUNT(EMP.ENAME)
FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DEPT.DNAME
HAVING COUNT(EMP.ENAME) BETWEEN 2 AND 6

15. 

SELECT EMP.*, DEPT.DNAME
FROM EMP 
INNER JOIN DEPT 
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.SAL = (SELECT MAX(SAL) FROM EMP)
